from flask import Flask, request, jsonify
from flask_cors import CORS
from gtts import gTTS
from deep_translator import GoogleTranslator
import os
import uuid

app = Flask(__name__)
CORS(app)

# Ensure 'static' directory exists
if not os.path.exists("static"):
    os.makedirs("static")

# Allowed accents mapping with their TLDs
TLD_MAP = {
    "us": "com",
    "uk": "co.uk",
    "india": "co.in",
    "australia": "com.au",
    "south_africa": "co.za"
}

@app.route("/speak", methods=["POST"])
def speak():
    try:
        text = request.json.get('text', '').strip()
        accent = request.json.get("accent", "us").lower()

        if not text:
            return jsonify({"error": "No text provided"}), 400

        # Use correct TLD
        tld = TLD_MAP.get(accent, "com")

        # Translate to Bengali
        translated = GoogleTranslator(source='auto', target='bn').translate(text)

        # Generate speech with selected accent
        tts = gTTS(text=text, lang='en', tld=tld)

        # Save audio to static folder
        filename = f"audio_{uuid.uuid4().hex}.mp3"
        filepath = os.path.join("static", filename)
        tts.save(filepath)

        return jsonify({
            "translation": translated,
            "audio_url": f"/static/{filename}"
        })

    except Exception as e:
        print("Error:", e)
        return jsonify({"error": "Translation or TTS failed"}), 500

if __name__ == "__main__":
    app.run(debug=True)
